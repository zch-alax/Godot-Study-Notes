# 类的实现方式
```GDScript
class Animal:
    extends Object # 如果不指定，默认继承Object
    const STATIC_FIELD = "静态变量"
    # 属性
    var height: int
    func _init():
        print("Animal类的构造方法")
    func move():
        print("自定义方法")
    static func staticFunc():
        print(STATIC_FIELD)

class Dog:
    extends Animal:
    func _init(height):
        self.height = height
        print("Dog类的构造方法")
    func move():
        print("用4条腿跑路")
        .move()
    func info():
        print("height:" + height as String)
```
